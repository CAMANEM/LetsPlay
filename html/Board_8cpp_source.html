182 void Board::A_Star(bool isTurn, int x, int y) {
183  isPlayerTurn = isTurn;
184  if(x != -1, y != -1){
185  start=&Grid[y*13+x];
186  }
187 
188  for (int x = 0; x < 13; x++){
189  for (int y = 0; y < 9; y++){
190  Grid[y*13 + x].beenVisited = false;
191  Grid[y*13 + x].Goal = std::numeric_limits<int>::max();
192  Grid[y*13 + x].LocalGoal = std::numeric_limits<int>::max();
193  Grid[y*13 + x].Previous = nullptr; // No parents
194  }
195  }
196 
197  auto distance = [](Squares* a, Squares* b) // For convenience
198  {
199  return sqrt((a->x - b->x)*(a->x - b->x) + (a->y - b->y)*(a->y - b->y));
200  };
201 
202  auto heuristic = [distance](Squares* a, Squares* b) // So we can experiment with heuristic
203  {
204  return distance(a, b);
205  };
206 
207  Squares *nodeCurrent = start;
208  start->LocalGoal = 0.0f;
209  start->Goal = heuristic(start, end);
210 
211  std::list<Squares*> listNotTestedNodes;
212  listNotTestedNodes.push_back(start);
213 
214  while(!listNotTestedNodes.empty()){
215  listNotTestedNodes.sort([](const Squares* lhs, const Squares* rhs){ return lhs->Goal < rhs->Goal; } );
216  while(!listNotTestedNodes.empty() && listNotTestedNodes.front()->beenVisited){
217  listNotTestedNodes.pop_front();
218  } if (listNotTestedNodes.empty()) {
219  break;
220  }
221 
222  nodeCurrent = listNotTestedNodes.front();
223  nodeCurrent->beenVisited = true;
224 
225  for (auto nodeNeighbour : nodeCurrent->Neighbours){
226  if (!nodeNeighbour->beenVisited && nodeNeighbour->isObstacle == 0) {
227  listNotTestedNodes.push_back(nodeNeighbour);
228  }
229 
230  float fPossiblyLowerGoal = nodeCurrent->LocalGoal + distance(nodeCurrent, nodeNeighbour);
231 
232  if (fPossiblyLowerGoal < nodeNeighbour->LocalGoal){
233  nodeNeighbour->Previous = nodeCurrent;
234  nodeNeighbour->LocalGoal = fPossiblyLowerGoal;
235 
236  nodeNeighbour->Goal = nodeNeighbour->LocalGoal + heuristic(nodeNeighbour, end);
237  }
238  }
239  }
