122 void Puzzle_Window::fragmentImage(int pieces) {
123 
124  // Calcular si se puede generar en una matriz nxn
125  int i_sqrt = sqrt(pieces);
126  float f_sqrt = sqrt(pieces);
127  int limit_i;
128 
129  int fragments_width;
130  int fragments_height;
131 
132  if (i_sqrt == f_sqrt){
133  fragments_width = sprite.getGlobalBounds().width / i_sqrt;
134  fragments_height = sprite.getGlobalBounds().height / i_sqrt;
135  limit_i = i_sqrt;
136  }
137  // Solo se divide en 2
138  else if (pieces == 2){
139  fragments_width = sprite.getGlobalBounds().width;
140  fragments_height = sprite.getGlobalBounds().height/2;
141  limit_i = 1;
142  }
143  // Numero a dividir es par
144  else if (pieces%2 == 0){
145  fragments_width = sprite.getGlobalBounds().width / 2;
146  limit_i = 2;
147  for (int i = 2; i < pieces; ++i) {
148  if (2*i == pieces){
149  fragments_height = sprite.getGlobalBounds().height/i;
150  break;
151  }
152  }
153  }
154  // Numero a fragmentar es impar
155  else{
156  fragments_width = sprite.getGlobalBounds().width;
157  fragments_height = sprite.getGlobalBounds().height/pieces;
158  limit_i = 1;
159  }
160 
161  sf::Vector2f position = sprite.getPosition();
162  sf::Vector2f sector = sf::Vector2f(0.0f, 0.0f);
163 
164  int i = 1;
165 
166  while (puzzle_pieces.size() < pieces){
167  sf::Sprite piece;
168  piece.setPosition(position);
169  piece.setTexture(texture);
170  piece.setTextureRect(sf::IntRect(sector.x, sector.y, fragments_width, fragments_height));
171  pieces_positions.push_back(position);
172  puzzle_pieces.push_back(piece);
173 
174  if (i < limit_i){
175  i++;
176  position.x = position.x + fragments_width +1;
177  sector.x = sector.x + fragments_width;
178  }
179  else{
180  i = 1;
181  position = sf::Vector2(sprite.getPosition().x, position.y + fragments_height+1);
182  sector = sf::Vector2f(0.0f, sector.y + fragments_height);
183  }
184  }
185 }

304  const sf::Vector2u originalImageSize{ originalImage.getSize() };
305  const sf::Vector2u resizedImageSize{ resizedImage.getSize() };
