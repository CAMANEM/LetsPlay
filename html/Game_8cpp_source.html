98  sf::Vector2f s = line->getPoints()[0].position;
99  sf::Vector2f e = line->getPoints()[1].position;
100  sf::Vector2f ps = p - s;
101  sf::Vector2f se = e - s;
102  float lengthLine = (e.x - s.x) * (e.x - s.x) + (e.y - s.y) * (e.y - s.y);
103  float t = ((ps.x * se.x) + (ps.y * se.y)) / lengthLine;
104  sf::Vector2f st;
105  st.x = s.x + t*se.x;
106  st.y = s.y + t*se.y;
107 
108  sf::Vector2f distance = p - st;
109  float distanceBetween = sqrtf((distance.x*distance.x) + (distance.y*distance.y));
110 
111  sf::Vector2f normal = distance / distanceBetween;
112  float dotProductNormal = circle->getVelocity().x*normal.x +
113  circle->getVelocity().y*normal.y;
114 
115  sf::Vector2f tangential = sf::Vector2f(-normal.y, normal.x);
116  float dotProductTangential = circle->getVelocity().x*tangential.x +
117  circle->getVelocity().y * tangential.y;
118 
119  float overlap = distanceBetween - circle->getRadius();
120  if (distanceBetween <= circle->getRadius())
121  {
122  if (t > -0.f && t < 1.f)
123  {
124  circle->setPosition(p.x - distance.x * overlap / distanceBetween,
125  p.y -distance.y * overlap / distanceBetween);
